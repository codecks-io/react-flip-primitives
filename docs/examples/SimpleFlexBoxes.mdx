import {Value} from "react-powerplug";
import FlipGroup from "../../src/index";

<Value initial={1}>
  {({value, set}) => (
    <FlipGroup
      noAnimationOnMount
      changeKey={value}
      durationMs={1500}
      keysAndData={Array.from(new Array(value), (_, index) => ({
        key: index.toString(),
        data: 40,
      }))}
      enterPositionStyle={{width: 1}}
      enterDecorationStyle={{opacity: 0}}
      leaveStyle={{opacity: 0, width: 1}}
    >
      {(registerNode, keysAndData) => (
        <React.Fragment>
          <div style={{display: "flex", flexWrap: "wrap"}}>
            {keysAndData.map(({key, data: number}) => (
              <div
                key={key}
                style={{
                  width: `${number}%`,
                  height: 50,
                  backgroundColor: `hsl(${number * 5}, 50%, 50%)`,
                  margin: "0 10px 20px",
                  opacity: 1,
                }}
                ref={registerNode(key, {
                  transitionProps: ["background-color", "opacity"],
                })}
              />
            ))}
          </div>
          <div ref={registerNode("buttons")}>
            <button onClick={() => set(value + 1)}>+1</button>
            <button onClick={() => set(value - 1)}>-1</button>
          </div>
        </React.Fragment>
      )}
    </FlipGroup>
  )}
</Value>
